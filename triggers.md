# SECUENCIAS
## Plantilla 
```sql
CREATE SEQUENCE TABLE_NAME_SEQ INCREMENT BY 1 START WITH 1 MAXVALUE 999 MINVALUE 1 NOCACHE;
```

# Triggers
## Plantilla 
```sql
CREATE OR REPLACE TRIGGER TABLE_NAME_TRG 
  BEFORE INSERT OR UPDATE ON TABLE_NAME 
  FOR EACH ROW 
BEGIN 
  IF INSERTING THEN 
    :NEW.CREATED    :=SYSDATE;
    :NEW.CREATED_BY := NVL(V('APP_USER'),USER);
    :NEW.STATUS := 1;
    IF :NEW.ID IS NULL THEN
      SELECT TABLE_NAME_SEQ.NEXTVAL INTO :NEW.ID FROM SYS.DUAL;
    END IF;
  END IF;
  
  IF UPDATING THEN
    :NEW.UPDATED    :=SYSDATE;
    :NEW.UPDATED_BY := NVL(V('APP_USER'),USER);
  END IF;
END;
```
